252 boost::asio::awaitable<Result<Lease>> acquire(
253 Endpoint endpoint, std::chrono::steady_clock::duration timeout =
254 std::chrono::steady_clock::duration::max()) {
255 normalize_endpoint(endpoint);
256
257 for (;;) {
258 // Fast path: try without allocating a waiter
259 if (auto lease = try_acquire(endpoint)) {
260 m_metrics.acquire_success.fetch_add(
261 1, std::memory_order_relaxed);
262 co_return Result<Lease>::ok(std::move(*lease));
263 }
264
265 // Check if we are shutting down
266 if (!m_state->alive.load(std::memory_order_acquire)) {
267 m_metrics.acquire_shutdown.fetch_add(
268 1, std::memory_order_relaxed);
269 co_return Result<Lease>::err(
270 Error{Error::Code::Unknown, "Pool is shutting down"});
271 }
272
273 auto timer = std::make_shared<boost::asio::steady_timer>(m_ex);
274
275 timer->expires_after(timeout);
276
277 std::list<Waiter>::iterator waiter_iterator;
278 {
279 std::lock_guard<std::mutex> lock(mu_);
280
281 // Re-check shutdown under lock
282 if (!m_state->alive.load(std::memory_order_acquire)) {
283 m_metrics.acquire_shutdown.fetch_add(
284 1, std::memory_order_relaxed);
285 co_return Result<Lease>::err(Error{
286 Error::Code::Unknown, "Pool is shutting down"});
287 }
288
289 // Determine wait reason and enqueue waiter
290 WaitReason reason = determine_wait_reason_locked(endpoint);
291
292 // Enqueue into primary list
293 waiter_iterator = m_waiters.emplace(m_waiters.end(),
294 Waiter{.ep = endpoint,
295 .reason = reason,
296 .timer = timer,
297 .active = true,
298 .queue_it = {}});
299
300 // Enqueue into secondary list
301 if (reason == WaitReason::EndpointCapacity) {
302 auto& bucket = m_buckets[endpoint];
303 bucket.waiters.push_back(&(*waiter_iterator));
304 waiter_iterator->queue_it =
305 std::prev(bucket.waiters.end());
306 } else {
307 m_global_waiters.push_back(&(*waiter_iterator));
308 waiter_iterator->queue_it =
309 std::prev(m_global_waiters.end());
310 }
311
312 m_metrics.waiters_total.fetch_add(
313 1, std::memory_order_relaxed);
314
315 // Close lost-wakeup window
316 if (auto lock_con_2 = try_acquire_locked(endpoint)) {
317 // Remove from queues if we acquired immediately
318 if (reason == WaitReason::EndpointCapacity) {
319 m_buckets[endpoint].waiters.erase(
320 waiter_iterator->queue_it);
321 } else {
322 m_global_waiters.erase(waiter_iterator->queue_it);
323 }
324
325 m_waiters.erase(waiter_iterator);
326 m_metrics.waiters_total.fetch_sub(
327 1, std::memory_order_relaxed);
328 m_metrics.acquire_success.fetch_add(
329 1, std::memory_order_relaxed);
330 co_return Result<Lease>::ok(std::move(*lock_con_2));
331 }
332 }
333
334 boost::system::error_code errc;
335 co_await timer->async_wait(boost::asio::redirect_error(
336 boost::asio::use_awaitable, errc));
337
338 // Remove ourselves on ALL exit paths
339 {
340 std::lock_guard<std::mutex> lock(mu_);
341 // ALWAYS erase the waiter, regardless of active status
342 // The waiter is ours, and we must remove it to keep the
343 // list clean
344
345 // If we are still active (timeout/cancel), we also need to
346 // remove from the secondary queue because pop_waiter didn't
347 // do it.
348 if (waiter_iterator->active) {
349 try {
350 if (waiter_iterator->reason ==
351 WaitReason::EndpointCapacity) {
352 // Important: Check if bucket still exists?
353 // Buckets are usually only created, rarely
354 // destroyed from map.
355 auto bucket_iter = m_buckets.find(endpoint);
356 if (bucket_iter != m_buckets.end()) {
357 bucket_iter->second.waiters.erase(
358 waiter_iterator->queue_it);
359 }
360 } else {
361 m_global_waiters.erase(
362 waiter_iterator->queue_it);
363 }
364 // NOLINTNEXTLINE(bugprone-empty-catch)
365 } catch (...) {
366 // Should not happen with valid iterators
367 }
368 m_metrics.waiters_total.fetch_sub(
369 1, std::memory_order_relaxed);
370 }
371
372 m_waiters.erase(waiter_iterator);
373 }
374
375 if (!errc) {
376 // Timer expired = timeout
377 m_metrics.acquire_timeout.fetch_add(
378 1, std::memory_order_relaxed);
379 co_return Result<Lease>::err(
380 Error{Error::Code::Timeout, "Acquire timeout"});
381 }
382
383 if (errc == boost::asio::error::operation_aborted) {
384 continue;
385 }
386
387 // Unexpected error
388 m_metrics.acquire_internal_error.fetch_add(
389 1, std::memory_order_relaxed);
390 co_return Result<Lease>::err(Error{
391 Error::Code::Unknown, "Internal error: " + errc.message()});
392 }
393 }

431 std::chrono::steady_clock::duration timeout) {

433
434 while (std::chrono::steady_clock::now() < deadline) {

532 void check_invariants_locked() {
533#ifndef NDEBUG
534 std::size_t computed_total = 0;
535
536 for (auto const& [ep, b] : m_buckets) {
537 computed_total += b.in_use.size();
538
539 // Invariant: no connection in both idle and in_use
540 std::unordered_set<Conn const*> in_use_ptrs;
541 for (auto const& [id, up] : b.in_use) {
542 assert(up && "in_use connection is null");
543 in_use_ptrs.insert(up.get());
544 }
545
546 for (auto const& entry : b.idle) {
547 assert(entry.conn && "idle connection is null");
548 assert(in_use_ptrs.find(entry.conn.get()) ==
549 in_use_ptrs.end() &&
550 "connection in both idle and in_use");
551 }
552 }
553
554 assert(computed_total == m_total_in_use && "m_total_in_use drift");
555#else
556#endif
557 }

560 WaitReason determine_wait_reason_locked(
561 Endpoint const& endpoint) const {
562 auto buck_iterator = m_buckets.find(endpoint);
563 if (buck_iterator == m_buckets.end()) {
564 // New endpoint, check global capacity
565 std::size_t total_total = m_total_in_use + total_idle_locked();
566 return (total_total >= m_cfg.max_total_connections)
567 ? WaitReason::GlobalCapacity
568 : WaitReason::EndpointCapacity;
569 }
570
571 auto const& bucket = buck_iterator->second;
572 std::size_t ep_total = bucket.in_use.size() + bucket.idle.size();
573
574 if (ep_total >= m_cfg.max_connections_per_endpoint) {
575 return WaitReason::EndpointCapacity;
576 }
577 return WaitReason::GlobalCapacity;
578 }

581 std::shared_ptr<boost::asio::steady_timer>
582 pop_waiter_for_endpoint_locked(Endpoint const& endpoint) {
583 // First: try endpoint-specific waiters (O(1))
584 auto buck_iterator = m_buckets.find(endpoint);
585 if (buck_iterator != m_buckets.end()) {
586 auto& bucket = buck_iterator->second;
587 while (!bucket.waiters.empty()) {
588 Waiter* waiter = bucket.waiters.front();
589 bucket.waiters.pop_front(); // Remove from secondary queue
590
591 if (waiter->active) {
592 waiter->active = false;
593 auto timer = waiter->timer;
594 m_metrics.waiters_total.fetch_sub(
595 1, std::memory_order_relaxed);
596 return timer;
597 }
598 }
599 }
600
601 // Second: try "any endpoint" / global waiters (O(1))
602 while (!m_global_waiters.empty()) {
603 Waiter* global_waiter = m_global_waiters.front();
604 m_global_waiters.pop_front(); // Remove from secondary queue
605
606 if (global_waiter->active) {
607 global_waiter->active = false;
608 auto timer = global_waiter->timer;
609 m_metrics.waiters_total.fetch_sub(
610 1, std::memory_order_relaxed);
611 return timer;
612 }
613 }
614
615 return {};
616 }

628 void prune_idle_locked(std::chrono::steady_clock::time_point now) {
629 if (m_cfg.connection_idle_ttl.count() <= 0) {
630 return;
631 }
632
633 for (auto& [_endpoint, _bucket] : m_buckets) {
634 while (!_bucket.idle.empty()) {
635 auto const& front = _bucket.idle.front();
636 if (now - front.last_used < m_cfg.connection_idle_ttl) {
637 break;
638 }
639
640 if (m_cfg.close_on_prune && front.conn) {
641 front.conn->close_http();
642 front.conn->close_https();
643 }
644 _bucket.idle.pop_front();
645 m_metrics.connection_pruned.fetch_add(
646 1, std::memory_order_relaxed);
647 }

699 void release(Endpoint const& endpoint, std::uint64_t ident) noexcept {
700 std::shared_ptr<boost::asio::steady_timer> timer;
701
702 try {
703 std::lock_guard<std::mutex> lock(mu_);
704 auto bucket_iter = m_buckets.find(endpoint);
705
706 if (bucket_iter == m_buckets.end()) {
707 m_metrics.release_invalid_id.fetch_add(
708 1, std::memory_order_relaxed);
709 return;
710 }
711
712 auto& bucket = bucket_iter->second;
713 auto it2 = bucket.in_use.find(ident);
714
715 if (it2 == bucket.in_use.end()) {
716 m_metrics.release_invalid_id.fetch_add(
717 1, std::memory_order_relaxed);
718 return;
719 }
720
721 auto uptr_conn = std::move(it2->second);
722 bucket.in_use.erase(it2);
723 --m_total_in_use;
724 m_metrics.total_in_use.store(m_total_in_use,
725 std::memory_order_relaxed);
726
727 // Health check before recycling
728 bool healthy = uptr_conn && is_connection_healthy(*uptr_conn);
729
730 if (healthy) {
731 auto now = std::chrono::steady_clock::now();
732 bucket.idle.push_back(IdleEntry{
733 .conn = std::move(uptr_conn),
734 .last_used = now,
735 .created = now, // created time (will be set properly
736 // on first create)
737 .reuse_count = 0 // reuse_count
738 });
739 m_metrics.total_idle.fetch_add(1,
740 std::memory_order_relaxed);
741 } else {
742 m_metrics.connection_dropped_unhealthy.fetch_add(
743 1, std::memory_order_relaxed);
744 // Connection dropped, capacity freed
745 }
746
747 // Pop waiter UNDER LOCK, but don't cancel just yet
748 timer = pop_waiter_for_endpoint_locked(endpoint);
749
750 check_invariants_locked();
751 } catch (...) {
752 // Swallow all exceptions in release to avoid throwing from
753 // destructor
754 }
755
756 // Cancel OUTSIDE lock to avoid deadlock
757 if (timer) {
758 timer->cancel();
759 }
760 }

763 std::optional<Lease> try_acquire_locked(Endpoint const& endpoint) {
764 // Check shutdown
765 if (!m_state->alive.load(std::memory_order_acquire)) {
766 return std::nullopt;
767 }
768
769 auto now = std::chrono::steady_clock::now();
770 prune_idle_locked(now);
771
772 auto& bucket = m_buckets[endpoint];
773
774 // Check circuit breaker
775 if (bucket.is_circuit_open(now)) {
776 return std::nullopt;
777 }
778
779 // Prefer reusing idle connections
780 while (!bucket.idle.empty()) {
781 auto entry = std::move(bucket.idle.front());
782 bucket.idle.pop_front();
783 m_metrics.total_idle.fetch_sub(1, std::memory_order_relaxed);
784
785 // Validate health
786 if (!entry.conn || !is_connection_healthy(*entry.conn)) {
787 m_metrics.connection_dropped_unhealthy.fetch_add(
788 1, std::memory_order_relaxed);
789 continue; // Try next idle connection
790 }
791
792 // Check reuse limit
793 if (entry.reuse_count >= m_cfg.max_connection_reuse_count) {
794 m_metrics.connection_dropped_reuse_limit.fetch_add(
795 1, std::memory_order_relaxed);
796 continue;
797 }
798
799 // Check age limit
800 if (now - entry.created > m_cfg.max_connection_age) {
801 m_metrics.connection_dropped_age_limit.fetch_add(
802 1, std::memory_order_relaxed);
803 continue;
804 }
805
806 // Connection is good to reuse
807 entry.reuse_count++;
808 auto ident = m_next_id++;
809 Conn* raw = entry.conn.get();
810 bucket.in_use.emplace(ident, std::move(entry.conn));
811 ++m_total_in_use;
812
813 m_metrics.total_in_use.store(m_total_in_use,
814 std::memory_order_relaxed);
815 m_metrics.connection_reused.fetch_add(
816 1, std::memory_order_relaxed);
817
818 check_invariants_locked();
819 return Lease(
820 m_state, raw, endpoint, ident,
821 [this](Endpoint const& endpoint, std::uint64_t id_) {
822 release(endpoint, id_);
823 });
824 }
825
826 // No idle connections available, check capacity for creating new
827 const std::size_t endpoint_total =
828 bucket.in_use.size() + bucket.idle.size();
829 const std::size_t total_total =
830 m_total_in_use + total_idle_locked();
831
832 if (endpoint_total >= m_cfg.max_connections_per_endpoint) {
833 return std::nullopt;
834 }
835 if (total_total >= m_cfg.max_total_connections) {
836 return std::nullopt;
837 }
838
839 // Create a new connection and mark it in-use
840 auto connection_instance =
841 std::make_unique<Conn>(m_ex, m_ssl_ctx, endpoint);
842 Conn* raw = connection_instance.get();
843 auto connection_id = m_next_id++;
844
845 bucket.in_use.emplace(connection_id,
846 std::move(connection_instance));
847 ++m_total_in_use;
848
849 m_metrics.total_in_use.store(m_total_in_use,
850 std::memory_order_relaxed);
851 m_metrics.connection_created.fetch_add(1,
852 std::memory_order_relaxed);
853
854 check_invariants_locked();
855 return Lease(m_state, raw, endpoint, connection_id,
856 [this](Endpoint const& endpoint, std::uint64_t id_) {
857 release(endpoint, id_);
858 });
859 }
