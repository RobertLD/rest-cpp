49 inline std::string url_encode(std::string_view s) {
50 static const char* hex = "0123456789ABCDEF";
51 std::string out;
52 out.reserve(s.size());
53 for (unsigned char c : s) {
54 if (std::isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~') {
55 out.push_back(c);
56 } else {
57 out.push_back('%');
58 out.push_back(hex[c >> 4]);
59 out.push_back(hex[c & 15]);

122 inline Result<UrlComponents> parse_url(const std::string& url) {
123 auto make_err = [&](std::string msg) -> Result<UrlComponents> {
124 Error e{};
125 e.message = std::move(msg);
126 e.code = Error::Code::InvalidUrl;
127 return Result<UrlComponents>::err(std::move(e));
128 };
129
130 std::string_view s(url);
131
132 bool https = false;
133 if (s.rfind("https://", 0) == 0) {
134 https = true;
135 s.remove_prefix(std::string_view("https://").size());
136 } else if (s.rfind("http://", 0) == 0) {
137 https = false;
138 s.remove_prefix(std::string_view("http://").size());
139 } else {
140 return make_err("URL must start with http:// or https://");
141 }
142
143 // Split host[:port] from path
144 std::string_view hostport = s;
145 std::string_view path = "/";
146 if (auto slash = s.find('/'); slash != std::string_view::npos) {
147 hostport = s.substr(0, slash);
148 path = s.substr(slash);
149 }
150
151 if (hostport.empty()) {
152 return make_err("URL missing host");
153 }
154
155 std::string host;
156 std::string port;
157
158 // Port parsing (simple: last ':' splits host/port)
159 if (auto colon = hostport.rfind(':'); colon != std::string_view::npos) {
160 host = std::string(hostport.substr(0, colon));
161 port = std::string(hostport.substr(colon + 1));
162 if (port.empty()) {
163 return make_err("URL has empty port");
164 }
165 } else {
166 host = std::string(hostport);
167 port = https ? "443" : "80";
168 }
169
170 if (host.empty()) {
171 return make_err("URL has empty host");
172 }
173
174 UrlComponents out;
175 out.https = https;
176 out.host = std::move(host);
177 out.port = std::move(port);
178 out.target = path.empty() ? "/" : std::string(path);
179 return Result<UrlComponents>::ok(std::move(out));
180 }

186 inline Result<UrlComponents> parse_base_url(std::string_view base_url) {
187 auto make_err = [&](std::string msg) -> Result<UrlComponents> {
188 Error e{};
189 e.message = std::move(msg);
190 e.code = Error::Code::InvalidUrl;
191 return Result<UrlComponents>::err(std::move(e));
192 };
193
194 if (base_url.empty()) {
195 return make_err("base_url is empty");
196 }
197 if (!is_absolute_url_with_protocol(base_url)) {
198 return make_err("base_url must start with http:// or https://");
199 }
200
201 auto parsed = rest_cpp::parse_url(std::string(base_url));
202 if (parsed.has_error())
203 return Result<UrlComponents>::err(parsed.error());
204
205 UrlComponents b = std::move(parsed.value());
206
207 // Normalize prefix in b.target
208 b.target = trim_trailing_slashes(std::move(b.target));
209 if (b.target == "/") b.target.clear();
210
211 // Keep joining simple/fast: reject query in base prefix
212 if (b.target.find('?') != std::string::npos) {
213 return make_err("base_url must not include query parameters");
214 }
215
216 return Result<UrlComponents>::ok(std::move(b));
217 }

219 inline Result<UrlComponents> resolve_url(std::string_view uri_or_url,
220 const UrlComponents* base) {
221 auto make_err = [&](std::string msg) -> Result<UrlComponents> {
222 Error e{};
223 e.message = std::move(msg);
224 e.code = Error::Code::InvalidUrl;
225 return Result<UrlComponents>::err(std::move(e));
226 };
227
228 // Absolute URL => parse (slow path)
229 if (is_absolute_url_with_protocol(uri_or_url)) {
230 return rest_cpp::parse_url(std::string(uri_or_url));
231 }
232
233 // Relative => requires base
234 if (base == nullptr || base->host.empty() || base->port.empty()) {
235 return make_err("Relative URI provided but base_url is empty");
236 }
237
238 // Normalize rel: "" => "/", "health" => "/health"
239 std::string_view rel = uri_or_url;
240 std::string rel_storage;
241
242 if (rel.empty()) {
243 rel = "/";
244 } else if (rel.front() != '/') {
245 rel_storage.reserve(rel.size() + 1);
246 rel_storage.push_back('/');
247 rel_storage.append(rel);
248 rel = rel_storage;
249 }
250
251 // Join prefix(base->target) + rel
252 std::string target;
253 if (base->target.empty()) {
254 target.assign(rel);
255 } else {
256 target.reserve(base->target.size() + rel.size());
257 target.append(base->target); // prefix has no trailing '/'
258 target.append(rel); // rel begins with '/'
259 }
260
261 UrlComponents out;
262 out.https = base->https;
263 out.host = base->host;
264 out.port = base->port;
265 out.target = std::move(target);
266 return Result<UrlComponents>::ok(std::move(out));
267 }
